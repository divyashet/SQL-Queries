--Action
				SELECT DISTINCT A.ACTION_CODE,
				  A.ACTION_NAME,
				  D.NAME ORGANIZATION_NAME,
				  to_char(A.START_DATE,'YYYY-MM-DD') START_DATE,
				  to_char(A.END_DATE,'YYYY-MM-DD') END_DATE,
				  B.ACTION_TYPE_CODE,
				  C.MEANING
				  --B.CREATED_BY
				FROM FUSION.PER_ACTIONS_VL A,
				  FUSION.PER_ACTIONS_B B,
				  FUSION.PER_ACTION_TYPES_TL C,
				  FUSION.HR_ORGANIZATION_V D
				WHERE A.ACTION_ID       =B.ACTION_ID
				AND A.BUSINESS_GROUP_ID = D.ORGANIZATION_ID
				AND B.ACTION_TYPE_ID    =C.ACTION_TYPE_ID;
--Action Reason
				SELECT DISTINCT A.ACTION_REASON_CODE,
				  C.NAME ORGANIZATION_NAME,
				  B.ACTION_REASON,
				  TO_CHAR(A.START_DATE,'YYYY-MM-DD') START_DATE,
				  TO_CHAR(A.END_DATE,'YYYY-MM-DD') END_DATE
				  --B.CREATED_BY
				FROM FUSION.PER_ACTION_REASONS_B A,
				  FUSION.PER_ACTION_REASONS_TL B,
				  FUSION.HR_ORGANIZATION_V C
				WHERE A.ACTION_REASON_ID=B.ACTION_REASON_ID
				AND A.BUSINESS_GROUP_ID = C.ORGANIZATION_ID;
--Business Unit
				SELECT DISTINCT A.NAME ORGANIZATION_NAME,
				  TO_CHAR(A.EFFECTIVE_START_DATE,'YYYY-MM-DD') EFFECTIVE_START_DAT,
				  TO_CHAR(A.EFFECTIVE_END_DATE,'YYYY-MM-DD') EFFECTIVE_END_DATE,
				  B.NAME ENTERPRISE_NAME,
				  DECODE(A.INTERNAL_EXTERNAL_FLAG,'INT','Internal','External') Internal_or_External,
				  C1.LOCATION_CODE,
				  C1.DESCRIPTION LOCATION_DESCRIPTION,
				  E.NAME LEGAL_ENTITY_NAME,
				  DECODE(A.STATUS,'A','ACTIVE','INACTIVE') STATUS,
				  F.SET_CODE,
				  F.SET_NAME,
				  H.DISPLAY_NAME MANAGER_NAME,
				  DECODE(G.ORG_INFORMATION6, 'Y', 'Yes','No') Enabled_for_HR
				FROM FUSION.HR_ORGANIZATION_V A,
				  FUSION.HR_ORGANIZATION_UNITS_F_TL B,
				  FUSION.PER_LOCATION_DETAILS_F C,
				  FUSION.PER_LOCATION_DETAILS_F_TL C1,
				  FUSION.HRT_ESTABLISHMENTS_B D,
				  FUSION.HR_ORGANIZATION_UNITS_F_TL E,
				  FUSION.FND_SETID_SETS F,
				  FUSION.HR_ORGANIZATION_INFORMATION_F G,
				  FUSION.PER_PERSON_NAMES_F H
				WHERE A.BUSINESS_GROUP_ID = B.ORGANIZATION_ID
				AND A.CLASSIFICATION_CODE = 'FUN_BUSINESS_UNIT'
				AND A.LOCATION_ID         = C.LOCATION_ID(+)
				AND C.LOCATION_DETAILS_ID = C1.LOCATION_DETAILS_ID(+)
				AND A.ESTABLISHMENT_ID    = D.ESTABLISHMENT_ID (+)
				AND A.ORGANIZATION_ID     = G.ORGANIZATION_ID (+)
				AND G.ORG_INFORMATION1    = H.PERSON_ID (+)
				AND A.LEGAL_ENTITY_ID     = E.ORGANIZATION_ID (+)
				AND G.ORG_INFORMATION4    = F.SET_ID (+)
				ORDER BY A.NAME;
